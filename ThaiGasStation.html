require("dotenv").config(); const express = require("express"); const axios = require("axios"); const cors = require("cors"); const path = require("path"); const app = express(); const PORT = process.env.PORT || 3000; const GOOGLE_API_KEY = process.env.GOOGLE_API_KEY || ""; app.use(cors()); app.use(express.json()); app.use(express.static(path.join(__dirname, "public"))); // ------------------------------- // Utility // ------------------------------- function haversineKm(lat1, lon1, lat2, lon2) { const R = 6371; const dLat = ((lat2 - lat1) * Math.PI) / 180; const dLon = ((lon2 - lon1) * Math.PI) / 180; const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos((lat1 * Math.PI) / 180) * Math.cos((lat2 * Math.PI) / 180) * Math.sin(dLon / 2) * Math.sin(dLon / 2); return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)); } function safeNum(value) { const n = Number(value); return Number.isFinite(n) ? n : null; } function dedupeStations(stations) { const map = new Map(); for (const s of stations) { const name = (s.name || "").trim().toLowerCase(); const lat = safeNum(s.lat); const lng = safeNum(s.lng); if (lat == null || lng == null) continue; const key = `${name}|${lat.toFixed(4)}|${lng.toFixed(4)}`; if (!map.has(key)) { map.set(key, s); } } return Array.from(map.values()); } // ------------------------------- // Overpass: Thailand all fuel stations // ------------------------------- async function fetchThailandFuelStationsFromOSM() { const query = ` [out:json][timeout:120]; area["ISO3166-1"="TH"][admin_level=2]->.th; ( node["amenity"="fuel"](area.th); way["amenity"="fuel"](area.th); relation["amenity"="fuel"](area.th); ); out center tags; `; const url = "https://overpass-api.de/api/interpreter"; const response = await axios.post(url, query, { headers: { "Content-Type": "text/plain" }, timeout: 180000 }); const elements = response.data?.elements || []; const stations = elements .map((el) => { const lat = el.lat ?? el.center?.lat; const lng = el.lon ?? el.center?.lon; return { source: "osm", osmType: el.type, osmId: el.id, name: el.tags?.name || "Unknown Station", brand: el.tags?.brand || "", operator: el.tags?.operator || "", address: [ el.tags?.["addr:housenumber"], el.tags?.["addr:street"], el.tags?.["addr:subdistrict"], el.tags?.["addr:district"], el.tags?.["addr:province"] ] .filter(Boolean) .join(" ") || "", lat, lng, openingHours: el.tags?.opening_hours || "", phone: el.tags?.phone || "", website: el.tags?.website || "", rawTags: el.tags || {} }; }) .filter((s) => safeNum(s.lat) != null && safeNum(s.lng) != null); return dedupeStations(stations); } // ------------------------------- // Google Places Nearby Search // ------------------------------- async function fetchNearbyGoogleStations(lat, lng, radius = 5000, maxResultCount = 20) { if (!GOOGLE_API_KEY) { return []; } const url = "https://places.googleapis.com/v1/places:searchNearby"; const body = { includedTypes: ["gas_station"], maxResultCount, locationRestriction: { circle: { center: { latitude: lat, longitude: lng }, radius } } }; const response = await axios.post(url, body, { headers: { "Content-Type": "application/json", "X-Goog-Api-Key": GOOGLE_API_KEY, "X-Goog-FieldMask": "places.id,places.displayName,places.formattedAddress,places.location,places.primaryType" }, timeout: 30000 }); const places = response.data?.places || []; return places .map((p) => ({ source: "google", googlePlaceId: p.id, name: p.displayName?.text || "Unknown Station", address: p.formattedAddress || "", lat: p.location?.latitude, lng: p.location?.longitude, primaryType: p.primaryType || "" })) .filter((s) => safeNum(s.lat) != null && safeNum(s.lng) != null); } // ------------------------------- // Google Place Details // ------------------------------- async function fetchGooglePlaceDetails(placeId) { if (!GOOGLE_API_KEY || !placeId) { return null; } const url = `https://places.googleapis.com/v1/places/${placeId}`; const response = await axios.get(url, { headers: { "X-Goog-Api-Key": GOOGLE_API_KEY, "X-Goog-FieldMask": "id,displayName,formattedAddress,location,fuelOptions,regularOpeningHours,nationalPhoneNumber,websiteUri" }, timeout: 30000 }); const p = response.data || {}; return { source: "google", googlePlaceId: p.id, name: p.displayName?.text || "Unknown Station", address: p.formattedAddress || "", lat: p.location?.latitude, lng: p.location?.longitude, fuelOptions: p.fuelOptions || null, regularOpeningHours: p.regularOpeningHours || null, nationalPhoneNumber: p.nationalPhoneNumber || "", websiteUri: p.websiteUri || "" }; } // ------------------------------- // API: health // ------------------------------- app.get("/api/health", (req, res) => { res.json({ ok: true, service: "fuel-th-app", googleConfigured: !!GOOGLE_API_KEY, timestamp: new Date().toISOString() }); }); // ------------------------------- // API: stations/all-th // ดึงปั๊มทั้งประเทศไทยจาก OSM // ------------------------------- app.get("/api/stations/all-th", async (req, res) => { try { const stations = await fetchThailandFuelStationsFromOSM(); res.json({ ok: true, count: stations.length, source: "osm-overpass", stations }); } catch (error) { console.error("all-th error:", error.message); res.status(500).json({ ok: false, error: "Failed to fetch Thailand fuel stations", detail: error.message }); } }); // ------------------------------- // API: stations/nearby // query: // lat=13.75 // lng=100.50 // radius=5000 // limit=20 // ------------------------------- app.get("/api/stations/nearby", async (req, res) => { try { const lat = safeNum(req.query.lat); const lng = safeNum(req.query.lng); const radius = safeNum(req.query.radius) || 5000; const limit = safeNum(req.query.limit) || 20; if (lat == null || lng == null) { return res.status(400).json({ ok: false, error: "lat and lng are required" }); } const googleStations = await fetchNearbyGoogleStations(lat, lng, radius, limit); const withDistance = googleStations .map((s) => ({ ...s, distanceKm: haversineKm(lat, lng, s.lat, s.lng) })) .sort((a, b) => a.distanceKm - b.distanceKm); res.json({ ok: true, count: withDistance.length, source: "google-places", stations: withDistance }); } catch (error) { console.error("nearby error:", error.response?.data || error.message); res.status(500).json({ ok: false, error: "Failed to fetch nearby stations", detail: error.response?.data || error.message }); } }); // ------------------------------- // API: stations/details/:placeId // ------------------------------- app.get("/api/stations/details/:placeId", async (req, res) => { try { const placeId = req.params.placeId; const details = await fetchGooglePlaceDetails(placeId); if (!details) { return res.status(404).json({ ok: false, error: "Place not found or Google API not configured" }); } res.json({ ok: true, source: "google-place-details", place: details }); } catch (error) { console.error("details error:", error.response?.data || error.message); res.status(500).json({ ok: false, error: "Failed to fetch place details", detail: error.response?.data || error.message }); } }); // ------------------------------- // API: aggregate/nearby-with-details // nearby + details พร้อม fuelOptions ถ้ามี // ------------------------------- app.get("/api/aggregate/nearby-with-details", async (req, res) => { try { const lat = safeNum(req.query.lat); const lng = safeNum(req.query.lng); const radius = safeNum(req.query.radius) || 5000; const limit = safeNum(req.query.limit) || 10; if (lat == null || lng == null) { return res.status(400).json({ ok: false, error: "lat and lng are required" }); } const nearby = await fetchNearbyGoogleStations(lat, lng, radius, limit); const detailed = []; for (const station of nearby) { try { const details = await fetchGooglePlaceDetails(station.googlePlaceId); detailed.push({ ...station, distanceKm: haversineKm(lat, lng, station.lat, station.lng), details }); } catch (e) { detailed.push({ ...station, distanceKm: haversineKm(lat, lng, station.lat, station.lng), details: null, detailError: e.message }); } } detailed.sort((a, b) => a.distanceKm - b.distanceKm); res.json({ ok: true, count: detailed.length, stations: detailed }); } catch (error) { console.error("aggregate error:", error.response?.data || error.message); res.status(500).json({ ok: false, error: "Failed to aggregate nearby stations", detail: error.response?.data || error.message }); } }); app.listen(PORT, () => { console.log(`Server running on http://localhost:${PORT}`); });
